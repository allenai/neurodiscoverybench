```python
from scipy.stats import chi2_contingency

# Extract relevant columns
adnc_column = 'Overall AD neuropathological Change'
cognitive_status_column = 'Cognitive Status'
dx_columns = [
    'Consensus Clinical Dx (choice=Alzheimers disease)',
    'Consensus Clinical Dx (choice=Alzheimers Possible/ Probable)',
    'Consensus Clinical Dx (choice=Dementia with Lewy Bodies/ Lewy Body Disease)'  # Considered dementia
]

# Create a flag for dementia presence

def is_dementia(row):
    return (row[cognitive_status_column] == 'Dementia') or any(row[dx] == 'Checked' for dx in dx_columns)

dataset['Has Dementia'] = dataset.apply(is_dementia, axis=1)

# Count total cases and dementia cases by ADNC category
adnc_dementia_counts = dataset.groupby(adnc_column)['Has Dementia'].agg(['sum', 'size']).rename(columns={'sum': 'Dementia Cases', 'size': 'Total Cases'})
adnc_dementia_counts['Dementia Prevalence'] = adnc_dementia_counts['Dementia Cases'] / adnc_dementia_counts['Total Cases']

# Output the dementia prevalence by ADNC
print(adnc_dementia_counts)

# Prepare data for chi-square test
contingency_table = adnc_dementia_counts[['Dementia Cases', 'Total Cases']].values
contingency_table[:, 1] -= contingency_table[:, 0]  # Adjust to show non-dementia cases vs dementia cases

# Perform chi-square test
chi2, p, _, _ = chi2_contingency(contingency_table)

print(f"\nChi-square test results: Chi2 = {chi2:.4f}, p-value = {p:.4f}")
```
