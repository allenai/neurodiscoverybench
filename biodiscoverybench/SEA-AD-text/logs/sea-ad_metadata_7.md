from scipy.stats import spearmanr

late_stage_ordinal = df['LATE'].map({'LATE Stage 1': 0, 'LATE Stage 2': 1, 'LATE Stage 3': 2, 'LATE Stage 4': 3, 'LATE Stage 5': 4})

# Ensure that all these mappings were correctly applied and handle 'NaN' cases if mapping failed
df['ADNC_Ordinal'] = adnc_ordinal

correlation_results = {}

# Spearman correlation calculation and significance testing
# Correlate ADNC with each of the comorbid neuropathologies
correlation_results['ADNC vs Highest Lewy Body Disease'] = spearmanr(df['ADNC_Ordinal'], lewy_body_disease_ordinal, nan_policy='omit')

# Print correlation results

    print(f"{key} - Spearman Correlation: {result.correlation}, p-value: {result.pvalue}")
